## Static styles

```tsx
import { tw } from 'typewritingclass'

function Card({ children }: { children: React.ReactNode }) {
  return <div className={tw.p(6).bg('blue-500').textColor('white').rounded('lg').hover.bg('blue-600')}>{children}</div>
}
```

## Dynamic values with useStyle()

```tsx
import { useStyle } from 'typewritingclass-react'
import { p, bg, rounded, dynamic } from 'typewritingclass'

function Banner({ color }: { color: string }) {
  const props = useStyle(p(6), bg(dynamic(color)), rounded('lg'))
  return <div {...props}>Welcome!</div>
}
```

## Next.js App Router

```tsx
// app/layout.tsx
import { getStyleTag } from 'typewritingclass-react/server'

export default function RootLayout({ children }: { children: React.ReactNode }) {
  return (
    <html lang="en">
      <head dangerouslySetInnerHTML={{ __html: getStyleTag() }} />
      <body>{children}</body>
    </html>
  )
}
```

Server Components use `tw` or `cx()` directly. Client Components with dynamic values use `useStyle()` with `'use client'`.
