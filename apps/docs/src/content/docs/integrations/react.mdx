## Static styles

React requires a string for `className`. Use a template literal to coerce `tw` chains:

```tsx
import { tw } from 'typewritingclass'

function Card({ children }: { children: React.ReactNode }) {
  return <div className={`${tw.p(6).bg.blue500.textColor.white.rounded.lg.hover.bg.blue600}`}>{children}</div>
}
```

## Dynamic values with useStyle()

```tsx
import { useStyle } from 'typewritingclass-react'
import { p, bg, rounded, dynamic } from 'typewritingclass'

function Banner({ color }: { color: string }) {
  const props = useStyle(p(6), bg(dynamic(color)), rounded.lg)
  return <div {...props}>Welcome!</div>
}
```
